#!/usr/bin/env bash
# Re-runnable probe log for flywheel-rhdcq BLOCKED disposition.
# Each surface emits exactly one BLOCKED_SURFACE_<N> marker on stdout.
# Exit 0 always — this is a probe, not a gate.

set +e

export PATH="/usr/bin:/bin:/usr/local/bin:/opt/homebrew/bin:$HOME/.local/bin:$HOME/.cargo/bin"

# ---- Surface 1 + 2: doctrine-sync.sh scope is L-rules (not docs) AND
#                    canonical-source regex returns L0 (sharded layout regression).
VS=$(bash /Users/josh/Developer/flywheel/.flywheel/scripts/doctrine-sync.sh --version-stamp 2>/dev/null)
echo "version-stamp: $VS"
if echo "$VS" | grep -qE '"highest_l_rule":[[:space:]]*"L0"'; then
  echo "BLOCKED_SURFACE_2 doctrine-sync L0 regression (sharded AGENTS.md has no inline ## L<N> headers)"
fi
USAGE=$(bash /Users/josh/Developer/flywheel/.flywheel/scripts/doctrine-sync.sh --unknown-arg 2>&1 | head -3)
if echo "$USAGE" | grep -q -- '--l-rules'; then
  echo "BLOCKED_SURFACE_1 doctrine-sync.sh propagates L-rules only (not .flywheel/doctrine/*.md docs)"
fi

# ---- Surface 3: canonical-doctrine-sync.sh referenced but not shipped.
HITS=$(find /Users/josh/Developer /Users/josh/.claude/skills /Users/josh/.local -maxdepth 6 -name 'canonical-doctrine-sync*' -type f 2>/dev/null | wc -l | tr -d ' ')
echo "canonical-doctrine-sync find count: $HITS"
if [ "$HITS" = "0" ]; then
  echo "BLOCKED_SURFACE_3 canonical-doctrine-sync.sh missing (referenced in skillos:1 WAVE-2 handoff but not shipped)"
fi

# ---- Surface 4: xref-skillos stub pattern absent from flywheel doctrine/.
XHITS=$(grep -lE 'xref-skillos|canonical-locator: skillos|STRICT.MIRROR|mirror-of-skillos|source: skillos' /Users/josh/Developer/flywheel/.flywheel/doctrine/*.md 2>/dev/null | wc -l | tr -d ' ')
echo "xref-skillos pattern matches: $XHITS"
if [ "$XHITS" = "0" ]; then
  echo "BLOCKED_SURFACE_4 9 xref-skillos stub files do not exist in .flywheel/doctrine/"
fi

# ---- Surface 5: target list has holes.
MISSING=()
for r in alpsinsurance mobile-eats clutterfreespaces picoz vrtx terratitle blackfoot; do
  if [ ! -d "/Users/josh/Developer/$r" ]; then
    MISSING+=("$r:absent")
  elif [ ! -f "/Users/josh/Developer/$r/.flywheel/STATE.json" ]; then
    MISSING+=("$r:no_state_json")
  fi
done
echo "target hole count: ${#MISSING[@]} (${MISSING[*]:-none})"
if [ "${#MISSING[@]}" -gt 0 ]; then
  echo "BLOCKED_SURFACE_5 target list holes (${MISSING[*]})"
fi

exit 0
