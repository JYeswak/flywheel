repo_docs_state() {
    local missing=0 needs_review=0 drift=0 p
    # FM-10 fix (2026-04-30, bd-cwfs2 Step 3): use frontmatter parser if shared
    # helper is available, fall back to body-grep only when missing. The
    # legacy body-grep tripped on STATE.md table rows merely mentioning the
    # trip-phrase (incident 2026-04-30T19:00Z).
    local _shared="${HOME}/.claude/hooks/_shared/frontmatter.sh"
    local use_frontmatter=0
    if [[ -r "$_shared" ]]; then
        # shellcheck source=/dev/null
        if source "$_shared" 2>/dev/null && declare -f frontmatter_status >/dev/null 2>&1; then
            use_frontmatter=1
        fi
    fi
    for p in "$(repo_state_path)" "$(repo_goal_path)" "$(repo_mission_path)"; do
        [[ -f "$p" ]] || { missing=1; continue; }
        if [[ "$use_frontmatter" -eq 1 ]]; then
            local status
            status="$(frontmatter_status "$p" 2>/dev/null)"
            case "$status" in
                locked)
                    local lock_hash body_hash
                    lock_hash="$(frontmatter_lock_hash "$p" 2>/dev/null)"
                    body_hash="$(frontmatter_body_sha256 "$p")"
                    if [[ -z "$lock_hash" || "$lock_hash" != "$body_hash" ]]; then
                        drift=1
                    fi
                    ;;
                ready) ;;
                *) needs_review=1 ;;
            esac
        else
            grep -q 'needs_owner_review' "$p" 2>/dev/null && needs_review=1
        fi
    done
    if [[ "$missing" -eq 1 ]]; then
        printf 'missing'
    elif [[ "$drift" -eq 1 ]]; then
        printf 'drift_detected'
    elif [[ "$needs_review" -eq 1 ]]; then
        printf 'needs_owner_review'
    else
        printf 'ready'
    fi
}
